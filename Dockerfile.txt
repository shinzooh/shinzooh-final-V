# 1. ابدأ من صورة بايثون رسمية ومستقرة
FROM python:3.11-slim

# 2. حدد مجلد العمل داخل الحاوية
WORKDIR /app

# 3. انسخ ملف التبعيات أولاً للاستفادة من ميزة التخزين المؤقت في Docker
COPY requirements.txt .

# 4. قم بتثبيت التبعيات باستخدام pip
RUN pip install --no-cache-dir -r requirements.txt

# 5. انسخ باقي ملفات الكود إلى مجلد العمل
COPY . .

# 6. عرّف الأمر الذي سيتم تشغيله عند بدء تشغيل الخدمة
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "10000"]
